\begin{enumerate}
\item
Let's use
\indented{
$\pA$ for ``$x$ is a person'',\\
$\pB$ for ``$x$ is lonely'', and\\
$\pC$ for ``$y$ is a friend of $x$''.
}
\begin{enumerate}
\item
Use $\pA$, $\pB$, and $\pC$ to express the statement
\indented{\textit{Every person is either lonely or has a friend}}
in terms of quantifiers.

\item
Now negate your answer, and push the $\neg$ all the way in.
\item
Finally, write out the result in English again.
Your sentence should intuitively sound like it means the same thing as 
``\textit{It is not the case that every person is either lonely or has a friend.}''
\end{enumerate}
\item
Let's use
\indented{
$\pA$ for ``$\epsilon$ is a library'',\\
$\pB$ for ``$\delta$ is a book in $\epsilon$'',\\
$\pC$ for ``$x$ is a page of $\delta$'', and\\
$\pD$ for ``$x$ mentions the name of $\epsilon$''.
}
\begin{enumerate}
\item
Use $\pA$, $\pB$, $\pC$, and $\pD$ to express the statement
\indented{\textit{There is a library in which every book has at least one page that does not mention the name of the library}}
in terms of quantifiers.

\item
Now negate your answer, and push the $\neg$ all the way in.

\item
Finally, write out the result in English again.
Your sentence should intuitively sound like it means the same thing as 
``\textit{It is not the case that there is a library 
in which every book has at least one page that does not mention the name of the library.}''
\end{enumerate}
\end{enumerate}
